реактивные формы (Reactive Forms) — это один из двух основных способов работы с формами в Angular (второй — Template-driven forms).
```typescript
import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-login',
  template: `
    <form [formGroup]="form" (ngSubmit)="onSubmit()">
      <input formControlName="email" placeholder="Email">
      <input formControlName="password" placeholder="Пароль" type="password">
      <button type="submit">Войти</button>
    </form>

    <p>Значение формы: {{ form.value | json }}</p>
  `,
})
export class LoginComponent {
  form = new FormGroup({
    email: new FormControl('', [Validators.required, Validators.email]),
    password: new FormControl('', [Validators.required, Validators.minLength(6)]),
  });

  onSubmit() {
    console.log(this.form.value);
  }
}

```
